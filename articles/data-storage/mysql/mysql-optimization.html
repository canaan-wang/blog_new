MySQL 性能优化

一、SQL 优化

避免 SELECT *

-- 不推荐
SELECT * FROM user WHERE id = 1;

-- 推荐
SELECT id, name, email FROM user WHERE id = 1;

分页优化

-- 深度分页性能差
SELECT * FROM user ORDER BY id LIMIT 100000, 10;

-- 优化1：使用覆盖索引
SELECT * FROM user 
WHERE id >= (SELECT id FROM user ORDER BY id LIMIT 100000, 1) 
LIMIT 10;

-- 优化2：记住上次位置
SELECT * FROM user WHERE id > last_id ORDER BY id LIMIT 10;

批量操作

-- 不推荐：循环单条插入
INSERT INTO user VALUES (1, 'a');
INSERT INTO user VALUES (2, 'b');

-- 推荐：批量插入
INSERT INTO user VALUES (1, 'a'), (2, 'b'), (3, 'c');

二、表结构优化

选择合适的数据类型

类型建议
INT主键、状态等整数
BIGINT大数据量 ID、时间戳
VARCHAR变长字符串
DECIMAL金额（避免浮点）
DATETIME日期时间

三、连接优化

-- 确保连接字段有索引
SELECT u.name, o.order_no 
FROM user u 
JOIN orders o ON u.id = o.user_id  -- user_id 需有索引
WHERE u.id = 1;

-- 小表驱动大表
SELECT * FROM small_table s 
JOIN big_table b ON s.id = b.small_id;

四、服务器配置优化

[mysqld]
# 连接数
max_connections = 500
max_connect_errors = 1000

# InnoDB 缓冲池（通常设为内存的 50-70%）
innodb_buffer_pool_size = 4G
innodb_buffer_pool_instances = 4

# 日志文件大小
innodb_log_file_size = 512M
innodb_log_buffer_size = 64M

# 刷盘策略（0/1/2）
innodb_flush_log_at_trx_commit = 2

# 查询缓存（MySQL 8.0 已移除）
query_cache_type = 1
query_cache_size = 64M

# 临时表大小
tmp_table_size = 64M
max_heap_table_size = 64M

五、慢查询分析

-- 开启慢查询日志
SET GLOBAL slow_query_log = 'ON';
SET GLOBAL long_query_time = 1;
SET GLOBAL log_queries_not_using_indexes = 'ON';

-- 查看慢查询
SELECT * FROM mysql.slow_log ORDER BY start_time DESC LIMIT 10;

-- 使用 mysqldumpslow 分析
mysqldumpslow -s t -t 10 /var/lib/mysql/slow.log

六、分区表

-- 按日期范围分区
CREATE TABLE logs (
    id BIGINT PRIMARY KEY,
    log_time DATETIME,
    message TEXT
) PARTITION BY RANGE (YEAR(log_time)) (
    PARTITION p2022 VALUES LESS THAN (2023),
    PARTITION p2023 VALUES LESS THAN (2024),
    PARTITION p2024 VALUES LESS THAN (2025),
    PARTITION p_future VALUES LESS THAN MAXVALUE
);

-- 查询自动分区裁剪
SELECT * FROM logs WHERE log_time >= '2024-01-01'; -- 只扫描 p2024
