Docker 基础

一、Docker 简介

Docker 是容器化平台，将应用及其依赖打包成标准单元，实现快速部署和环境一致性。

二、基础命令

# 镜像操作
docker pull nginx:latest
docker images
docker rmi nginx
docker build -t myapp:1.0 .

# 容器操作
docker run -d -p 80:80 --name web nginx
docker ps
docker ps -a
docker stop web
docker start web
docker rm web
docker exec -it web /bin/bash

# 日志
docker logs web
docker logs -f web

三、Dockerfile

# 基础镜像
FROM openjdk:11-jre-slim

# 工作目录
WORKDIR /app

# 复制文件
COPY target/app.jar app.jar

# 环境变量
ENV JAVA_OPTS="-Xmx512m"

# 暴露端口
EXPOSE 8080

# 启动命令
ENTRYPOINT ["sh", "-c", "java $JAVA_OPTS -jar app.jar"]

四、Docker Compose

version: '3.8'
services:
  web:
    build: .
    ports:
      - "8080:8080"
    environment:
      - DB_HOST=db
    depends_on:
      - db
  
  db:
    image: mysql:8.0
    environment:
      MYSQL_ROOT_PASSWORD: secret
      MYSQL_DATABASE: mydb
    volumes:
      - db_data:/var/lib/mysql

volumes:
  db_data:

五、常用操作

# 数据卷
docker volume create myvol
docker run -v myvol:/data nginx

# 网络
docker network create mynet
docker run --network mynet nginx

# 清理
docker system prune      # 清理无用数据
docker image prune       # 清理无用镜像
