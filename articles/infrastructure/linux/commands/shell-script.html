二、条件判断

# 数字比较
if [ $a -eq $b ]; then  # 等于
if [ $a -ne $b ]; then  # 不等于
if [ $a -gt $b ]; then  # 大于
if [ $a -lt $b ]; then  # 小于

# 字符串比较
if [ "$str1" = "$str2" ]; then
if [ -z "$str" ]; then   # 空字符串
if [ -n "$str" ]; then   # 非空

# 文件判断
if [ -e file ]; then     # 存在
if [ -f file ]; then     # 普通文件
if [ -d dir ]; then      # 目录
if [ -r file ]; then     # 可读
if [ -w file ]; then     # 可写
if [ -x file ]; then     # 可执行

三、流程控制

if 语句

if [ condition ]; then
    # 代码
elif [ condition ]; then
    # 代码
else
    # 代码
fi

case 语句

case $var in
    pattern1)
        # 代码
        ;;
    pattern2|pattern3)
        # 代码
        ;;
    *)
        # 默认
        ;;
esac

循环

# for 循环
for i in 1 2 3 4 5; do
    echo $i
done

for ((i=0; i<10; i++)); do
    echo $i
done

for file in *.txt; do
    echo $file
done

# while 循环
while [ condition ]; do
    # 代码
done

# until 循环
until [ condition ]; do
    # 代码
done

# 读取文件
while read line; do
    echo $line
done < file.txt

四、函数

# 定义函数
function greet() {
    local name=$1  # local 局部变量
    echo "Hello, $name!"
    return 0
}

# 调用
greet "Alice"
result=$?  # 获取返回值

# 返回值通过 echo
function add() {
    echo $(($1 + $2))
}
sum=$(add 5 3)

五、数组

# 定义数组
fruits=("apple" "banana" "cherry")

# 访问元素
echo ${fruits[0]}      # apple
echo ${fruits[@]}      # 所有元素
echo ${#fruits[@]}     # 数组长度

# 遍历
for fruit in "${fruits[@]}"; do
    echo $fruit
done

# 添加元素
fruits+=("date")

# 关联数组（Bash 4+）
declare -A user
user[name]="Alice"
user[age]=25
echo ${user[name]}

六、实用脚本示例

日志清理

#!/bin/bash
LOG_DIR="/var/log/app"
DAYS=7

find $LOG_DIR -name "*.log" -mtime +$DAYS -delete
echo "Cleaned logs older than $DAYS days"

批量重命名

#!/bin/bash
for file in *.txt; do
    [ -e "$file" ] || continue
    newname="backup_$file"
    mv "$file" "$newname"
    echo "Renamed: $file -> $newname"
done

服务监控

#!/bin/bash
SERVICE="nginx"

if ! pgrep -x "$SERVICE" > /dev/null; then
    echo "$SERVICE is not running, starting..."
    systemctl start $SERVICE
    echo "$SERVICE started at $(date)" >> /var/log/service_monitor.log
fi

数据库备份

#!/bin/bash
DB_NAME="mydb"
BACKUP_DIR="/backup"
DATE=$(date +%Y%m%d_%H%M%S)

mysqldump -u root -p'password' $DB_NAME > $BACKUP_DIR/${DB_NAME}_${DATE}.sql

# 保留最近 7 天
cd $BACKUP_DIR
ls -t ${DB_NAME}_*.sql | tail -n +8 | xargs -r rm
echo "Backup completed: ${DB_NAME}_${DATE}.sql"
