Linux 常用命令

一、文件操作

命令说明示例
ls列出目录ls -la, ls -lh
cd切换目录cd ~, cd -
pwd当前目录pwd
cp复制cp -r src dest
mv移动/重命名mv old new
rm删除rm -rf dir
touch创建空文件touch file.txt
mkdir创建目录mkdir -p a/b/c
cat查看文件cat file
less分页查看less file
head查看开头head -n 20 file
tail查看结尾tail -f log
find查找文件find . -name "*.java"
grep文本搜索grep -r "pattern" dir

二、系统监控

命令说明
top进程监控
htop增强版 top
ps查看进程 ps aux | grep java
free内存使用 free -h
df磁盘使用 df -h
du目录大小 du -sh dir
netstat网络连接 netstat -tlnp
ss替代 netstat ss -tlnp
iostatIO 统计
vmstat系统整体状态

三、文件权限

# 修改权限
chmod 755 file      # rwxr-xr-x
chmod u+x file      # 用户添加执行权限
chmod -R 755 dir    # 递归修改

# 修改所有者
chown user:group file
chown -R user:group dir

# 权限数字
# 4=读(r) 2=写(w) 1=执行(x)
# 7=rwx 6=rw- 5=r-x 4=r--

四、压缩解压

# tar
tar -czvf archive.tar.gz dir    # 压缩
tar -xzvf archive.tar.gz        # 解压
tar -tvf archive.tar.gz         # 查看内容

# zip
zip -r archive.zip dir
unzip archive.zip

# gzip
gzip file          # 压缩为 file.gz
gunzip file.gz     # 解压

五、网络命令

# 查看网络配置
ifconfig
ip addr

# 测试连通
ping google.com
curl -I http://example.com
wget url

# 端口测试
nc -zv host port
telnet host port

# DNS 查询
nslookup example.com

# 路由跟踪
traceroute example.com
mtr example.com

六、文本处理

# awk
awk '{print $1}' file           # 打印第一列
awk -F: '{print $1}' /etc/passwd
awk '$3 > 100 {print $0}' file  # 条件过滤

# sed
sed 's/old/new/g' file          # 替换文本
sed -i 's/old/new/g' file       # 直接修改文件
sed -n '10,20p' file            # 打印10-20行

# sort & uniq
sort file | uniq                # 去重
sort file | uniq -c             # 统计出现次数

# wc
wc -l file                      # 统计行数
wc -w file                      # 统计词数

# xargs
find . -name "*.log" | xargs rm -f
cat files.txt | xargs -I {} cp {} dest/

七、进程管理

# 前台转后台
./script.sh &
Ctrl+Z, then bg

# nohup
nohup ./script.sh > output.log 2>&1 &

# 查看作业
jobs
fg %1      # 前台运行作业1
bg %1      # 后台运行作业1

# 终止进程
kill PID
kill -9 PID    # 强制终止
killall java   # 终止所有 java 进程
pkill -f pattern
