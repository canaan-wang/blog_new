Linux 网络管理

一、网络配置

# 查看网络接口
ifconfig
ip addr
ip link

# 查看路由
route -n
ip route

# 查看 DNS
cat /etc/resolv.conf

# 配置 IP
ip addr add 192.168.1.100/24 dev eth0
ip link set eth0 up

二、网络诊断

# 测试连通
ping google.com

# 查看连接
netstat -tlnp
ss -tlnp

# 路由跟踪
traceroute google.com
mtr google.com

# DNS 查询
nslookup google.com
dig google.com
host google.com

# 端口扫描
nc -zv host 80
nmap -p 1-65535 localhost

三、抓包分析

# tcpdump
tcpdump -i eth0
tcpdump -i eth0 port 80
tcpdump -i eth0 host 192.168.1.1
tcpdump -i eth0 -w capture.pcap

# tshark
tshark -i eth0
tshark -r capture.pcap

四、防火墙

# iptables
iptables -L -n -v                    # 查看规则
iptables -A INPUT -p tcp --dport 22 -j ACCEPT  # 允许 SSH
iptables -A INPUT -p tcp --dport 80 -j ACCEPT  # 允许 HTTP
iptables -A INPUT -j DROP            # 丢弃其他

# firewalld
firewall-cmd --state
firewall-cmd --list-all
firewall-cmd --add-port=8080/tcp --permanent
firewall-cmd --reload

# ufw (Ubuntu)
ufw status
ufw allow 22/tcp
ufw enable

五、网络性能

# 带宽测试
iperf3 -s        # 服务端
iperf3 -c host   # 客户端

# 网络速度
iftop
nload

# 连接数统计
ss -s
netstat -ant | wc -l
