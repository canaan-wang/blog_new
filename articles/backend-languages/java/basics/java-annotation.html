Java 注解（Annotation）

一、内置注解

注解作用
@Override标记方法重写
@Deprecated标记已废弃
@SuppressWarnings抑制警告
@FunctionalInterface标记函数式接口
@SafeVarargs抑制可变参数警告

二、元注解

// 定义注解
@Target(ElementType.METHOD)        // 作用目标
@Retention(RetentionPolicy.RUNTIME) // 保留策略
@Documented                         // 包含在文档中
@Inherited                          // 子类继承
public @interface MyAnnotation {
    String value() default "";
    int count() default 0;
}

// 使用
@MyAnnotation(value = "test", count = 5)
public void myMethod() {}

三、@Target 取值

值说明
TYPE类、接口、枚举
FIELD字段、枚举常量
METHOD方法
PARAMETER参数
CONSTRUCTOR构造器
LOCAL_VARIABLE局部变量
ANNOTATION_TYPE注解
PACKAGE包

四、反射读取注解

// 获取类注解
MyAnnotation ann = clazz.getAnnotation(MyAnnotation.class);
if (ann != null) {
    System.out.println(ann.value());
}

// 获取方法注解
Method method = clazz.getMethod("test");
MyAnnotation methodAnn = method.getAnnotation(MyAnnotation.class);

// 获取字段注解
Field field = clazz.getDeclaredField("name");
MyAnnotation fieldAnn = field.getAnnotation(MyAnnotation.class);

// 获取所有注解
Annotation[] anns = clazz.getAnnotations();

五、实战：自定义日志注解

@Target(ElementType.METHOD)
@Retention(RetentionPolicy.RUNTIME)
public @interface Log {
    String value() default "";
    boolean printParams() default true;
}

// AOP 处理
@Aspect
@Component
public class LogAspect {
    
    @Around("@annotation(log)")
    public Object around(ProceedingJoinPoint point, Log log) throws Throwable {
        String methodName = point.getSignature().getName();
        long start = System.currentTimeMillis();
        
        if (log.printParams()) {
            System.out.println("Params: " + Arrays.toString(point.getArgs()));
        }
        
        try {
            return point.proceed();
        } finally {
            long cost = System.currentTimeMillis() - start;
            System.out.println(log.value() + " - " + methodName + " cost: " + cost + "ms");
        }
    }
}

// 使用
@Service
public class UserService {
    
    @Log("查询用户")
    public User getUser(Long id) {
        return userDao.findById(id);
    }
}

六、重复注解（Java 8+）

@Repeatable(Schedules.class)  // 指定容器注解
public @interface Scheduled {
    String cron();
}

public @interface Schedules {
    Scheduled[] value();
}

// 使用
@Scheduled(cron = "0 0 * * *")
@Scheduled(cron = "0 30 * * *")
public void task() {}
